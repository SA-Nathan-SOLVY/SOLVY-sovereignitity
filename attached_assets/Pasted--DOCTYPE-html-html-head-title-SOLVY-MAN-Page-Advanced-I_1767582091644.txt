    <script>
        // --- SAMPLE DATA & CONFIGURATION ---
        let config = {
            memberCount: 105,
            consumerVolume: 70000,
            consumerRate: 0.0165,  // 1.65%
            businessVolume: 30000,
            businessRate: 0.0230,  // 2.30%
            memberExamplePatronage: 0.08  // 8% for example member
        };

        // --- CORE CALCULATION FUNCTIONS ---
        function calculateRevenue(volume, rate) {
            return volume * rate;
        }
        
        function calculateTotalRevenue() {
            const consumerRev = calculateRevenue(config.consumerVolume, config.consumerRate);
            const businessRev = calculateRevenue(config.businessVolume, config.businessRate);
            return {
                consumer: consumerRev,
                business: businessRev,
                total: consumerRev + businessRev
            };
        }
        
        function allocateRevenue(totalRevenue) {
            return {
                ops: totalRevenue * 0.70,
                pool: totalRevenue * 0.20,
                fund: totalRevenue * 0.10
            };
        }
        
        function calculateBusinessContribution(businessRev, totalRev) {
            return totalRev > 0 ? (businessRev / totalRev) * 100 : 0;
        }
        
        function calculateBusinessBoost(businessRev, businessVol, consumerRate) {
            // What business volume would generate at consumer rates
            const atConsumerRate = businessVol * consumerRate;
            // The extra contribution from business rates
            return businessRev - atConsumerRate;
        }

        // --- DASHBOARD UPDATE FUNCTION ---
        function updateDashboard() {
            // Get current values from inputs
            config.consumerVolume = parseFloat(document.getElementById('consumerVolume').value) || 70000;
            config.consumerRate = (parseFloat(document.getElementById('consumerRate').value) || 1.65) / 100;
            config.businessVolume = parseFloat(document.getElementById('businessVolume').value) || 30000;
            config.businessRate = (parseFloat(document.getElementById('businessRate').value) || 2.30) / 100;
            
            // Perform all calculations
            const revenue = calculateTotalRevenue();
            const allocation = allocateRevenue(revenue.total);
            const businessContribution = calculateBusinessContribution(revenue.business, revenue.total);
            const businessBoost = calculateBusinessBoost(revenue.business, config.businessVolume, config.consumerRate);
            const memberShare = allocation.pool * config.memberExamplePatronage;
            
            const totalVolume = config.consumerVolume + config.businessVolume;
            const avgRate = totalVolume > 0 ? (revenue.total / totalVolume) * 100 : 0;
            
            // Update summary metrics
            document.getElementById('memberCount').textContent = config.memberCount.toLocaleString();
            document.getElementById('totalVolume').textContent = '$' + totalVolume.toLocaleString();
            document.getElementById('totalRevenue').textContent = '$' + revenue.total.toFixed(2);
            document.getElementById('businessContribution').textContent = businessContribution.toFixed(1) + '%';
            
            // Update table values
            document.getElementById('tableConsumerVol').textContent = config.consumerVolume.toLocaleString();
            document.getElementById('tableConsumerRate').textContent = (config.consumerRate * 100).toFixed(2);
            document.getElementById('tableConsumerRev').textContent = revenue.consumer.toFixed(2);
            document.getElementById('tableConsumerPool').textContent = (revenue.consumer * 0.20).toFixed(2);
            
            document.getElementById('tableBusinessVol').textContent = config.businessVolume.toLocaleString();
            document.getElementById('tableBusinessRate').textContent = (config.businessRate * 100).toFixed(2);
            document.getElementById('tableBusinessRev').textContent = revenue.business.toFixed(2);
            document.getElementById('tableBusinessPool').textContent = (revenue.business * 0.20).toFixed(2);
            
            document.getElementById('tableTotalVol').textContent = totalVolume.toLocaleString();
            document.getElementById('tableAvgRate').textContent = avgRate.toFixed(2);
            document.getElementById('tableTotalRev').textContent = revenue.total.toFixed(2);
            document.getElementById('tableTotalPool').textContent = allocation.pool.toFixed(2);
            
            // Update 70/20/10 allocation
            document.getElementById('opsAmount').textContent = '$' + allocation.ops.toFixed(2);
            document.getElementById('poolAmount').textContent = '$' + allocation.pool.toFixed(2);
            document.getElementById('fundAmount').textContent = '$' + allocation.fund.toFixed(2);
            document.getElementById('memberShare').textContent = '$' + memberShare.toFixed(2);
            document.getElementById('businessBoost').textContent = '$' + (businessBoost * 0.10).toFixed(2);
            
            // Update calculation breakdowns
            document.getElementById('businessCalc').textContent = 
                `(${config.businessVolume.toLocaleString()} × ${(config.businessRate*100).toFixed(2)}% = $${revenue.business.toFixed(2)})`;
            
            document.getElementById('opsCalc').textContent = 
                `$${revenue.total.toFixed(2)} × 0.70 = $${allocation.ops.toFixed(2)}`;
            document.getElementById('poolCalc').textContent = 
                `$${revenue.total.toFixed(2)} × 0.20 = $${allocation.pool.toFixed(2)}`;
            document.getElementById('fundCalc').textContent = 
                `$${revenue.total.toFixed(2)} × 0.10 = $${allocation.fund.toFixed(2)}`;
            document.getElementById('memberCalc').textContent = 
                `${(config.memberExamplePatronage*100).toFixed(0)}% of $${allocation.pool.toFixed(2)} = $${memberShare.toFixed(2)}`;
        }

        // --- INITIALIZE DASHBOARD ON PAGE LOAD ---
        document.addEventListener('DOMContentLoaded', function() {
            // Set initial input values
            document.getElementById('consumerVolume').value = config.consumerVolume;
            document.getElementById('consumerRate').value = (config.consumerRate * 100).toFixed(2);
            document.getElementById('businessVolume').value = config.businessVolume;
            document.getElementById('businessRate').value = (config.businessRate * 100).toFixed(2);
            
            // Run initial calculation
            updateDashboard();
            
            // Add event listeners for real-time updates (optional)
            const inputs = document.querySelectorAll('.revenue-input input');
            inputs.forEach(input => {
                input.addEventListener('change', updateDashboard);
            });
        });
    </script>
