<script>
(function() {
    const STORAGE_KEY = 'solvy_prelaunch_commitments';

    function loadCommitments() {
        try {
            const stored = localStorage.getItem(STORAGE_KEY);
            if (stored) return JSON.parse(stored);
        } catch(e) { console.warn(e); }
        return [
            { name: "Evergreen Mayo", email: "evergreen@ebl.beauty", pledge: 500, date: new Date().toISOString().slice(0,10), status: "committed" }
        ];
    }

    function saveCommitments(commitments) {
        localStorage.setItem(STORAGE_KEY, JSON.stringify(commitments));
    }

    function addCommitment(name, email, pledge) {
        const commitments = loadCommitments();
        commitments.push({
            name: name.trim(),
            email: email.trim(),
            pledge: parseFloat(pledge),
            date: new Date().toISOString().slice(0,10),
            status: "committed"
        });
        saveCommitments(commitments);
        renderTable();
        return commitments;
    }

    function renderTable() {
        const commitments = loadCommitments();
        const tbody = document.getElementById('commitmentTableBody');
        if (!tbody) return;
        if (commitments.length === 0) {
            tbody.innerHTML = '<tr><td colspan="5" style="text-align: center;">No commitments yet. Be the first.</td></tr>';
            document.getElementById('totalCommitmentDisplay').innerHTML = '';
            return;
        }
        let totalMonthly = 0;
        tbody.innerHTML = commitments.map(c => {
            totalMonthly += c.pledge;
            return `<tr>
                <td>${escapeHtml(c.name)}</td>
                <td>${escapeHtml(c.email)}</td>
                <td>$${c.pledge.toFixed(2)}</td>
                <td>${c.date}</td>
                <td><span class="status-badge">${c.status}</span></td>
            </tr>`;
        }).join('');
        document.getElementById('totalCommitmentDisplay').innerHTML = `💰 Total monthly committed volume: $${totalMonthly.toFixed(2)}<br>📈 Annual projected: $${(totalMonthly * 12).toFixed(2)}`;
    }

    function escapeHtml(str) { return str.replace(/[&<>]/g, function(m) { return m === '&' ? '&amp;' : m === '<' ? '&lt;' : '&gt;'; }); }

    function exportCSV() {
        const commitments = loadCommitments();
        if (!commitments.length) { alert("No commitments to export."); return; }
        const headers = ["Name", "Email", "Monthly Pledge (USD)", "Date", "Status"];
        const rows = commitments.map(c => [c.name, c.email, c.pledge, c.date, c.status]);
        const csv = [headers, ...rows].map(row => row.join(",")).join("\n");
        const blob = new Blob([csv], { type: "text/csv" });
        const a = document.createElement("a");
        a.href = URL.createObjectURL(blob);
        a.download = `solvy_commitments_${new Date().toISOString().slice(0,10)}.csv`;
        a.click();
        URL.revokeObjectURL(a.href);
    }

    document.getElementById('commitmentForm')?.addEventListener('submit', (e) => {
        e.preventDefault();
        const name = document.getElementById('memberName')?.value;
        const email = document.getElementById('memberEmail')?.value;
        const pledge = document.getElementById('pledgeAmount')?.value;
        if (!name || !email || !pledge) { alert("Please fill all fields."); return; }
        addCommitment(name, email, pledge);
        document.getElementById('commitmentForm').reset();
        alert(`Thank you, ${name}. Your commitment helps build the cooperative.`);
    });
    document.getElementById('exportBtn')?.addEventListener('click', exportCSV);
    renderTable();
})();
</script>
