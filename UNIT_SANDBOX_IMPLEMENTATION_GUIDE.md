## STEP 2: READY TO LAUNCH COMPONENT

### 2.1 What Is Ready To Launch?

Unit's "Ready To Launch" is a pre-built onboarding component that handles:
- KYC/KYB identity verification
- Document upload (ID, proof of address)
- Account application submission
- Real-time application status

**Benefits**:
- Reduces development time by 4-6 weeks
- Built-in compliance (OFAC, sanctions screening)
- Mobile-responsive
- White-label customizable

### 2.2 Implementation Options

### 2.3 Backend: Generate JWT Token

```javascript
// /api/unit/generate-token.js
const jwt = require('jsonwebtoken');

const generateUnitToken = (customerId) => {
  const payload = {
    sub: customerId,
    org: process.env.UNIT_ORG_ID,
    iat: Math.floor(Date.now() / 1000),
    exp: Math.floor(Date.now() / 1000) + (60 * 60) // 1 hour
  };
  
  return jwt.sign(payload, process.env.UNIT_API_TOKEN);
};

module.exports = { generateUnitToken };
```

## STEP 3: WEBHOOK ENDPOINTS

### 3.1 Required Webhook Events

Subscribe to these events in Unit dashboard:
- `application.approved` → Create account and card
- `application.denied` → Notify member, provide alternatives
- `account.created` → Update member record
- `account.closed` → Handle member exit
- `customer.created` → Log for analytics
- `card.created` → Send card shipping notification
- `card.activated` → Enable card in app
- `transaction.created` → Capture interchange for 70/20/10
- `payment.clearing` → Confirm ACH transfers

### 3.2 Webhook Handler Implementation

```javascript
// /api/webhooks/unit.js
const crypto = require('crypto');
const express = require('express');
const router = express.Router();

// Verify webhook signature
const verifyWebhookSignature = (req) => {
  const signature = req.headers['x-unit-signature'];
  const payload = JSON.stringify(req.body);
  
  const expectedSignature = crypto
    .createHmac('sha256', process.env.SOLVY_WEBHOOK_SECRET)
    .update(payload)
    .digest('hex');
  
  return crypto.timingSafeEqual(
    Buffer.from(signature),
    Buffer.from(expectedSignature)
  );
};

// Main webhook handler
router.post('/unit', async (req, res) => {
  // Verify signature
  if (!verifyWebhookSignature(req)) {
    return res.status(401).send('Invalid signature');
  }
  
  const { data, type } = req.body;
  
  try {
    switch (type) {
      case 'application.approved':
        await handleApplicationApproved(data);
        break;
        
      case 'application.denied':
        await handleApplicationDenied(data);
        break;
        
      case 'account.created':
        await handleAccountCreated(data);
        break;
        
      case 'card.activated':
        await handleCardActivated(data);
        break;
        
      case 'transaction.created':
        await handleTransactionCreated(data);
        break;
        
      default:
        console.log(`Unhandled webhook type: ${type}`);
    }
    
    res.status(200).send('OK');
  } catch (error) {
    console.error('Webhook processing error:', error);
    res.status(500).send('Error processing webhook');
  }
});

// Handler: Application Approved
const handleApplicationApproved = async (application) => {
  console.log('✅ Application approved:', application.id);
  
  // 1. Create deposit account
  const account = await createDepositAccount(application.relationships.customer.data.id);
  
  // 2. Create SOLVY card
  const card = await createSolvyCard(account.id);
  
  // 3. Update member record
  await db.members.update({
    unitApplicationId: application.id,
    unitCustomerId: application.relationships.customer.data.id,
    unitAccountId: account.id,
    unitCardId: card.id,
    status: 'approved',
    approvedAt: new Date()
  });
  
  // 4. Send welcome email
  await sendWelcomeEmail(member.email, card.lastFourDigits);
  
  console.log('✅ Member onboarding complete:', member.email);
};

// Handler: Transaction Created (Interchange Capture)
const handleTransactionCreated = async (transaction) => {
  // Only process card transactions (not ACH)
  if (transaction.type !== 'cardTransaction') return;
  
  console.log('💳 Transaction:', transaction.id, '$' + transaction.attributes.amount);
  
  // Get interchange amount from transaction
  const interchange = transaction.attributes.interchange || 0;
  
  if (interchange > 0) {
    // Apply 70/20/10 distribution
    const distribution = {
      transactionId: transaction.id,
      interchangeAmount: interchange,
      memberDividend: interchange * 0.70,
      communityPool: interchange * 0.20,
      operations: interchange * 0.10,
      memberId: transaction.relationships.account.data.id,
      createdAt: new Date()
    };
    
    // Store for quarterly distribution
    await db.distributions.create(distribution);
    
    console.log('💰 Interchange captured:', interchange, 'Distribution:', distribution);
  }
};

module.exports = router;
```

## STEP 4: CORE API IMPLEMENTATIONS

### 4.1 Unit Client Setup

```javascript
// /lib/unit.js
const axios = require('axios');

const unit = axios.create({
  baseURL: process.env.UNIT_API_URL,
  headers: {
    'Authorization': `Bearer ${process.env.UNIT_API_TOKEN}`,
    'Content-Type': 'application/vnd.api+json'
  }
});

// Add response interceptor for error handling
unit.interceptors.response.use(
  (response) => response.data,
  (error) => {
    console.error('Unit API Error:', error.response?.data || error.message);
    throw error;
  }
);

module.exports = unit;
```

### 4.2 Create Customer (Member)

```javascript
// /api/unit/create-customer.js
const unit = require('../../lib/unit');

const createCustomer = async (memberData) => {
  const response = await unit.post('/customers', {
    data: {
      type: 'individualCustomer',
      attributes: {
        ssn: memberData.ssn,
        fullName: {
          first: memberData.firstName,
          last: memberData.lastName
        },
        dateOfBirth: memberData.dateOfBirth,
        address: {
          street: memberData.address.street,
          city: memberData.address.city,
          state: memberData.address.state,
          postalCode: memberData.address.zip,
          country: 'US'
        },
        email: memberData.email,
        phone: {
          countryCode: '1',
          number: memberData.phone
        },
        tags: {
          memberId: memberData.id,
          cohort: 'first_circle',
          businessName: memberData.businessName
        }
      }
    }
  });
  
  return response.data;
};

module.exports = { createCustomer };
```
